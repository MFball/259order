// --- 強力連接版：針對芯燁打印機優化 --- async function handlePrintAndFinish() { // 如果還沒連接，先進行連接 if (!printCharacteristic) { try { // 1. 搜尋所有藍牙設備 (移除 XP- 名稱限制，避免搜不到) const device = await navigator.bluetooth.requestDevice({ acceptAllDevices: true, optionalServices: [ '0000ffe0-0000-1000-8000-00805f9b34fb', // 芯燁常用 1 '49535343-fe7d-4ae5-8fa9-9fafd205e455', // 芯燁常用 2 '000018f0-0000-1000-8000-00805f9b34fb' // 芯燁常用 3 ] }); alert("正在建立連接，請稍候..."); const server = await device.gatt.connect(); // 2. 嘗試尋找可用的服務 (自動偵測) let service; const serviceUUIDs = [ '0000ffe0-0000-1000-8000-00805f9b34fb', '49535343-fe7d-4ae5-8fa9-9fafd205e455', '000018f0-0000-1000-8000-00805f9b34fb' ]; for (let uuid of serviceUUIDs) { try { service = await server.getPrimaryService(uuid); if (service) break; } catch (e) { continue; } } if (!service) throw new Error("找不到打印服務 UUID"); // 3. 獲取寫入特徵值 const characteristics = await service.getCharacteristics(); // 尋找具有「寫入 (write)」權限的特徵 printCharacteristic = characteristics.find(c => c.properties.write || c.properties.writeWithoutResponse); if (!printCharacteristic) throw new Error("找不到可寫入的特徵值"); alert("藍牙打印機連接成功！請再次點擊『列印』出單。"); return; } catch (err) { console.error(err); alert("連接失敗詳情：" + err.message); return; } } // --- 以下為打印邏輯 --- try { // 初始化指令 (ESC/POS) let commands = [ 0x1B, 0x40, // 初始化 0x1B, 0x61, 0x01 // 居中 ]; // 點餐抬頭 let text = "Million Noodles\n\n"; cart.forEach((item, i) => { text += `#${i+1} ${item.names.join('+')} (${item.size})\n`; if(item.noodle) text += ` ${item.noodle}\n`; if(item.sauces.length) text += ` ${item.sauces.join(',')}\n`; text += "--------------------------------\n"; }); text += `TOTAL: $${document.getElementById('total-sum').innerText}\n\n\n\n\n`; const encoder = new TextEncoder(); const textData = encoder.encode(text); const fullCommands = new Uint8Array([...commands, ...textData, 0x1D, 0x56, 0x01]); await printCharacteristic.writeValue(fullCommands); // 清空數據 cart = []; updateCartDisplay(); resetForNewItem(); alert("打印完成！"); } catch (err) { alert("列印出錯：" + err.message); printCharacteristic = null; // 出錯時重置連接 } }
